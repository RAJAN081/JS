//strings and their fxns // 👉 Declare a string let name = "Raj the Coder"; // ----------------------------- // 📏 LENGTH - Count characters console.log(name.length); // 13 // 🔠 TOUPPERCASE - Make all letters capital console.log(name.toUpperCase()); // "RAJ THE CODER" // 🔡 TOLOWERCASE - Make all letters small console.log(name.toLowerCase()); // "raj the coder" // 🧍 CHARAT - Get a character at specific position console.log(name.charAt(0)); // "R" // 🔍 INCLUDES - Check if string contains another string console.log(name.includes("Coder")); // true // 🔍 INDEXOF - Find index of first match console.log(name.indexOf("C")); // 8 // ✂️ SLICE - Cut a piece of string (from, to) console.log(name.slice(0, 3)); // "Raj" console.log(name.slice(4)); // "the Coder" (from index 4 to end) // 🔁 REPLACE - Replace part of the string let edited = name.replace("Coder", "Legend"); console.log(edited); // "Raj the Legend" // ✨ TRIM - Remove spaces from both ends let messy = " I love JS "; console.log(messy.trim()); // "I love JS" // 🔪 SPLIT - Convert string to array based on separator let sentence = "JavaScript is power"; let words = sentence.split(" "); // Split by space console.log(words); // ["JavaScript", "is", "power"] // 🔗 CONCAT - Join strings together let first = "Hello"; let second = "World"; let joined = first.concat(" ", second); console.log(joined); // "Hello World" // 💬 TEMPLATE LITERALS - Embed variables in string using backticks `` let lang = "JavaScript"; console.log(`I code in ${lang}`); // "I code in JavaScript"
